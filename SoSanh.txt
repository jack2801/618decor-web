### 🔒 Các Cơ chế Khóa (Lockdown Mechanisms)
1.  **Sorting tuyệt đối:** Registry và Injection luôn được sắp xếp theo Alphabet `(A-Z)`. Dù ông nạp module Z trước module A, trong code cuối cùng A vẫn sẽ nằm trên Z.
2.  **No-Runtime-Seed:** Loại bỏ hàm `hash()` của Python (vốn đổi sau mỗi lần khởi động lại). Dùng `SHA256` cố định.
3.  **Atomic Swap:** Ghi file tạm $\rightarrow$ `os.replace` (Hệ điều hành đảm bảo tính nguyên vẹn). Không bao giờ có file lỗi.
4.  **State Fingerprint:** Sau mỗi hành động, Bot tính toán một mã Hash đại diện cho toàn bộ dự án. Nếu mã này đổi, nghĩa là dự án đã thay đổi.

INDEX_FILE = "Index.html"

MARKER_END = "<!-- MODULES_END -->"

def get_sha256(content: str) -> str:
    """Hàm băm cố định, không phụ thuộc seed của runtime"""
    return hashlib.sha256(content.encode('utf-8')).hexdigest()

def atomic_write(filepath: str, content: str) -> bool:
    """
    Ghi file an toàn tuyệt đối.
    1. Ghi vào file rác.
    2. Swap file rác vào vị trí chính thức.
    3. Không để lại tàn dư.
    """
    directory = os.path.dirname(filepath)
    if directory and not os.path.exists(directory):
        os.makedirs(directory)
        
    # Tạo file tạm, tắt auto-delete để control việc replace
    fd, tmp_path = tempfile.mkstemp(dir=directory, text=True)
    
    try:
        with os.fdopen(fd, 'w', encoding='utf-8', newline='\n') as f:
            f.write(content)
        
        # Nếu file đã tồn tại, check hash trước khi ghi đè (Idempotency)
        if os.path.exists(filepath):
            with open(filepath, 'r', encoding='utf-8') as f:
                old_content = f.read()
            if get_sha256(old_content) == get_sha256(content):
                os.remove(tmp_path) # Xóa file tạm vì không cần update
                return False # Không có thay đổi
        
        # Atomic Replace
        os.replace(tmp_path, filepath)
        return True # Có thay đổi
        
    except Exception as e:
        if os.path.exists(tmp_path):
            os.remove(tmp_path)
        raise e

class ProjectState:
    def __init__(self, root):
        self.root = root
        self.registry_path = os.path.join(root, REGISTRY_FILE)
        self.modules_path = os.path.join(root, MODULES_DIR)

    def get_registry(self) -> Dict[str, str]:
        """Đọc registry, nếu không có trả về dict rỗng"""
        if not os.path.exists(self.registry_path):
            return {}
        try:
            with open(self.registry_path, 'r', encoding='utf-8') as f:
                return json.load(f)
        except:
            return {}

    def save_registry(self, data: Dict[str, str]) -> bool:
        """
        Lưu registry với format JSON CỐ ĐỊNH từng byte.
        sort_keys=True: Đảm bảo thứ tự key luôn A-Z.
        indent=4: Format chuẩn.
        """
        # Sort data trước khi dump để đảm bảo JSON string là deterministic
        content = json.dumps(data, sort_keys=True, indent=4, ensure_ascii=False)
        return atomic_write(self.registry_path, content)

    def calculate_state_fingerprint(self) -> str:
        """
        Tính toán Hash của TOÀN BỘ PROJECT.
        Dùng để verify tính toàn vẹn và đồng nhất (Test 3).
        """
        hashes = []
        
        # 1. Hash Registry
        if os.path.exists(self.registry_path):
            with open(self.registry_path, 'r', encoding='utf-8') as f:
                hashes.append(f"REGISTRY:{get_sha256(f.read())}")
        
        # 2. Hash Index
        index_full = os.path.join(self.root, INDEX_FILE)
        if os.path.exists(index_full):
            with open(index_full, 'r', encoding='utf-8') as f:
                hashes.append(f"INDEX:{get_sha256(f.read())}")
                
        # 3. Hash All Modules (Sorted by filename)
        if os.path.exists(self.modules_path):
            files = sorted(os.listdir(self.modules_path))
            for f_name in files:
                if f_name.endswith(".js"):
                    f_path = os.path.join(self.modules_path, f_name)
                    with open(f_path, 'r', encoding='utf-8') as f:
                        hashes.append(f"MOD_{f_name}:{get_sha256(f.read())}")
        
        # Tổng hợp lại thành 1 chuỗi hash duy nhất
        full_state_str = "|".join(hashes)
        return get_sha256(full_state_str)

    def rebuild_index_injection(self):
        """
        Tạo lại khối <script> trong Index.html dựa trên Registry.
        Luôn sort theo Alphabet để đảm bảo tính Độc lập thứ tự (Order Independence).
        """
        index_full = os.path.join(self.root, INDEX_FILE)
        if not os.path.exists(index_full):
            return False

        with open(index_full, 'r', encoding='utf-8') as f:
            content = f.read()

        start_idx = content.find(MARKER_START)
        end_idx = content.find(MARKER_END)

        if start_idx == -1 or end_idx == -1:
            return False

        registry = self.get_registry()
        
        # --- LOGIC QUAN TRỌNG: SORT ALPHABET ---
        # Bất kể lúc thêm vào registry thứ tự nào, lúc inject luôn là A->Z
        sorted_modules = sorted(registry.items()) 
        
        injection_lines = []
        injection_lines.append(MARKER_START)
        for tag, filename in sorted_modules:
            # Đường dẫn chuẩn hóa
            path = f"{MODULES_DIR}/{filename}"
            injection_lines.append(f'    <script src="{path}"></script>')
        injection_lines.append(f"    {MARKER_END}")
        
        new_block = "\n".join(injection_lines)
        
        # Ghép lại file
        new_content = content[:start_idx] + new_block + content[end_idx + len(MARKER_END):]
        
        return atomic_write(index_full, new_content)

def process_clipboard(text: str, engine: ProjectState) -> str:
    """Xử lý Input -> Trả về Trace Log"""
    lines = text.splitlines()
    if not lines: return None
    
    first_line = lines[0].strip()
    match = re.match(r"^\[([A-Z0-9_]+)\]$", first_line)
    
    if not match: return None
    
    tag = match.group(1)
    content = "\n".join(lines[1:])
    trace_id = f"TX-{datetime.now().strftime('%H%M%S')}"
    
    log_msg = [f"[{trace_id}] 📥 INPUT: [{tag}]"]
    
    # 1. Xử lý logic file đích
    if tag == "INDEX":
        target_file = INDEX_FILE
        full_path = os.path.join(engine.root, target_file)
        # INDEX là trường hợp đặc biệt, ghi trực tiếp
        changed = atomic_write(full_path, content)
        if changed: log_msg.append(f"   + Write Index: OK")
        else: log_msg.append(f"   - Index unchanged (Idempotent)")
    else:
        # Module thường
        filename = f"{tag.lower()}.js"
        full_path = os.path.join(engine.modules_path, filename)
        
        # Bước 1: Ghi file module
        changed_file = atomic_write(full_path, content)
        if changed_file: log_msg.append(f"   + Write Module {filename}: OK")
        
        # Bước 2: Cập nhật Registry (Luôn sort)
        reg = engine.get_registry()
        if reg.get(tag) != filename:
            reg[tag] = filename
            engine.save_registry(reg)
            log_msg.append(f"   + Update Registry: OK")
        
        # Bước 3: Re-inject Index (Tự động sắp xếp lại)
        changed_idx = engine.rebuild_index_injection()
        if changed_idx: log_msg.append(f"   + Re-inject Index (Sorted): OK")

    # Final: Tính Fingerprint
    fp = engine.calculate_state_fingerprint()
    log_msg.append(f"   🔐 STATE HASH: {fp[:16]}...") # Chỉ hiện 16 ký tự đầu
    
    print("\n".join(log_msg))
    return fp

# --- MAIN LOOP ---

def stable_hash(data: str) -> str:
    """SHA-256 deterministic hash. KHÔNG BAO GIỜ dùng builtin hash()."""
    return hashlib.sha256(data.encode("utf-8")).hexdigest()

    """16-char prefix of SHA-256."""

def stable_order_key(tag: str) -> int:
    """
    Deterministic order cho unknown tags.
    Dùng SHA-256 thay vì hash() builtin.
    Kết quả: số nguyên 1000-1999, stable across runs.
    """
    h = stable_hash(tag)
    # Lấy 4 hex chars → int 0-65535 → mod 1000 → +1000
    return 1000 + (int(h[:4], 16) % 1000)

        modules.values(),

def file_bytes_hash(path: str) -> Optional[str]:
    """SHA-256 of file bytes. None if not exists."""
    if not os.path.exists(path):
        return None
    with open(path, "rb") as f:
        return hashlib.sha256(f.read()).hexdigest()


def compute_project_fingerprint(project_dir: str) -> str:
    """
    STATE FINGERPRINT: sha256 of sorted(all file hashes).
    Deterministic — không phụ thuộc filesystem order.
    """
    hashes = []
    for root, dirs, files in os.walk(project_dir):
        dirs.sort()  # deterministic walk order
        for fname in sorted(files):
            if fname.startswith("."):
                # Include registry but skip other hidden files
                if fname != CONFIG_FILE_NAME:
                    continue
            fpath = os.path.join(root, fname)
            rel = os.path.relpath(fpath, project_dir)
            fhash = file_bytes_hash(fpath)
            if fhash:
                hashes.append(f"{rel}:{fhash}")

    combined = "\n".join(sorted(hashes))
    return stable_hash(combined)

def parse_module_block(text: str) -> Union[ModuleData, Error]:
    """
    PURE. Parse clipboard text → ModuleData | Error.

    Quy tắc:
    - Dòng 1: [TAG]
    - Dòng 2 (optional nếu tag ∈ KNOWN_GROUPS): GROUP:name
    - Còn lại: JS body (phải có window.registerModule)
    - Không partial match. Không đoán. Sai → reject.
    """
    if not text or not text.strip():
        return Error(ErrorCode.PARSE_NO_TAG, "Empty input")

    lines = text.split("\n")
    first_line = lines[0].strip()

    # Step 1: Parse tag
    m = TAG_PATTERN.match(first_line)
    if not m:
        return Error(
            ErrorCode.PARSE_INVALID_TAG,
            f"Dòng đầu không hợp lệ: '{first_line}'",
            {"first_line": first_line},
        )
    tag = m.group(1)

    # Step 2: Parse group
    group: Optional[str] = None
    body_start = 1

    if len(lines) > 1:
        gm = GROUP_PATTERN.match(lines[1].strip())
        if gm:
            group = gm.group(1).lower()
            body_start = 2

    if group is None:
        if tag in KNOWN_GROUPS:
            group = tag.lower()
        else:
            return Error(
                ErrorCode.PARSE_MISSING_GROUP,
                f"[{tag}] không trong KNOWN_GROUPS và thiếu GROUP: ở dòng 2",
                {"tag": tag, "hint": "Thêm dòng 2: GROUP:decor"},
            )

    # Step 3: Body
    body = "\n".join(lines[body_start:]).strip()
    if not body:
        return Error(ErrorCode.PARSE_EMPTY_BODY, f"[{tag}] không có body", {"tag": tag})

    # Step 4: Metadata (deterministic)
    known = KNOWN_GROUPS.get(tag)
    if known:
        label = known["label"]
        icon = known["icon"]
        order = known["order"]
    else:
        label = tag.replace("_", " ").title()
        icon = "fas fa-puzzle-piece"
        order = stable_order_key(tag)  # SHA-256 based, NOT hash()

    return ModuleData(
        tag=tag, group=group, body=body,
        label=label, icon=icon, order=order,
        filename=f"{tag.lower()}.js",
    )


def validate_module(data: ModuleData) -> Union[bool, Error]:
    """PURE. Validate body content."""
    if len(data.body) < 50:
        return Error(
            ErrorCode.VALIDATE_BODY_TOO_SHORT,
            f"Body quá ngắn: {len(data.body)} chars (min 50)",
        )

    expected_a = f"window.registerModule('{data.page_name}'"
    expected_b = f'window.registerModule("{data.page_name}"'
    if expected_a not in data.body and expected_b not in data.body:
        return Error(
            ErrorCode.VALIDATE_NO_REGISTER_CALL,
            f"Thiếu: {expected_a}",
        )

    return True


def compute_file_paths(data: ModuleData, project_dir: str) -> PathMap:

def compute_script_tag(module_relative: str) -> str:
    """PURE."""
    return f'    <script src="{module_relative}"></script>'


def compute_index_injection(
    current_html: str,
    all_modules: Dict[str, RegistryEntry],
) -> Union[str, Error]:
    """
    PURE. Inject script tags vào index.html.

    Sort: group (alpha) → tag (alpha). KHÔNG phụ thuộc thứ tự thêm.
    Marker: phải đúng 1 cặp.
    Idempotent: inject 2 lần → kết quả giống nhau.
    """
    sc = current_html.count(INDEX_MARKER_START)
    ec = current_html.count(INDEX_MARKER_END)

    if sc == 0 or ec == 0:
        return Error(ErrorCode.FILE_MARKER_MISSING, "Thiếu marker", {"start": sc, "end": ec})
    if sc > 1 or ec > 1:
        return Error(ErrorCode.FILE_MARKER_DUPLICATE, "Trùng marker", {"start": sc, "end": ec})

    # DETERMINISTIC SORT: group → tag (alphabetic)
    sorted_mods = stable_sort_modules(all_modules)

    seen: Set[str] = set()
    lines = []
    for mod in sorted_mods:
        rel = f"{MODULES_DIR_NAME}/{mod.filename}"
        if rel not in seen:
            seen.add(rel)
            lines.append(compute_script_tag(rel))

    block = "\n".join(lines)
    replacement = f"{INDEX_MARKER_START}\n{block}\n{INDEX_MARKER_END}"

    pattern = re.escape(INDEX_MARKER_START) + r".*?" + re.escape(INDEX_MARKER_END)
    result = re.sub(pattern, replacement, current_html, flags=re.DOTALL)

    # Verify
    for mod in sorted_mods:
        tag_str = compute_script_tag(f"{MODULES_DIR_NAME}/{mod.filename}")
        if tag_str not in result:
            return Error(ErrorCode.INJECTION_DUPLICATE, f"Verify failed: {mod.filename}")

    return result


def compute_nav_html(all_modules: Dict[str, RegistryEntry]) -> str:
    """PURE. Generate nav HTML. Sort: group → tag."""
    sorted_mods = stable_sort_modules(all_modules)
    lines = []
    first = True
    for mod in sorted_mods:
        cls = ' class="nav-link active"' if first else ' class="nav-link"'
        first = False
        lines.append(
            f'                    <li><a href="#" '
            f'data-page="{mod.tag.lower()}"{cls}>{mod.label}</a></li>'
        )
    return "\n".join(lines)


def compute_registry_update(
    current: Registry, new_module: ModuleData, body_hash: str,
) -> Tuple[Registry, bool]:
    """
    PURE. Idempotent: same hash → no change.
    """
    existing = current.modules.get(new_module.tag)
    if existing and existing.content_hash == body_hash:
        return (current, False)

    new_reg = Registry(version=current.version)
    new_reg.modules = dict(current.modules)
    new_reg.modules[new_module.tag] = RegistryEntry(
        tag=new_module.tag, group=new_module.group,
        filename=new_module.filename, label=new_module.label,
        icon=new_module.icon, order=new_module.order,
        content_hash=body_hash,
    )
    return (new_reg, True)


def generate_file_operations(
    data: ModuleData,
    paths: PathMap,
    current_registry: Registry,
    current_index_html: str,
    existing_module_content: Optional[str],
) -> Union[List[FileOp], Error]:
    """
    PURE. Compute list of file ops.
    Idempotent: unchanged content → empty ops.
    """
    ops: List[FileOp] = []
    body_hash = stable_hash_short(data.body)

    # Dir
    ops.append(FileOp(
        FileOpType.CREATE_DIR,
        os.path.join(paths.project_dir, MODULES_DIR_NAME),
        description=f"Ensure {MODULES_DIR_NAME}/",
    ))

    # Module file
    if existing_module_content is not None:
        if stable_hash_short(existing_module_content) == body_hash:
            pass  # skip — identical
        else:
            ops.append(FileOp(
                FileOpType.WRITE_FILE, paths.module_file,
                data.body, f"Update {data.filename} (hash changed)",
            ))
    else:
        ops.append(FileOp(
            FileOpType.WRITE_FILE, paths.module_file,
            data.body, f"Create {data.filename}",
        ))

    # Registry
    new_reg, changed = compute_registry_update(current_registry, data, body_hash)

    if changed:
        # Index
        idx = compute_index_injection(current_index_html, new_reg.modules)
        if isinstance(idx, Error):
            return idx

        ops.append(FileOp(
            FileOpType.UPDATE_INDEX, paths.index_file,
            idx, f"Update index.html ({len(new_reg.modules)} modules)",
        ))
        ops.append(FileOp(
            FileOpType.UPDATE_REGISTRY, paths.registry_file,
            new_reg.to_json(), f"Update registry ({len(new_reg.modules)} entries)",
        ))

    return ops

def read_file(path: str) -> Optional[str]:
    try:
        with open(path, "r", encoding="utf-8") as f:
            return f.read()
    except FileNotFoundError:
        return None
    except Exception as e:
        LOG.error(f"Read failed: {path} → {e}")
        return None


def atomic_write(path: str, content: str) -> Union[bool, Error]:
    """
    Atomic write:
    1. Write to .tmp in SAME directory
    2. Verify hash
    3. os.replace() → atomic on all platforms
    4. NO .bak files. NO leftover on success.
    5. On failure → .tmp is cleaned up.
    """
    parent = os.path.dirname(path)
    os.makedirs(parent, exist_ok=True)

    # Use tempfile in same dir for atomic replace
    fd = None
    tmp_path = None
    try:
        fd, tmp_path = tempfile.mkstemp(dir=parent, suffix=".tmp")
        with os.fdopen(fd, "w", encoding="utf-8") as f:
            fd = None  # os.fdopen took ownership
            f.write(content)
            f.flush()
            os.fsync(f.fileno())

        # Verify
        with open(tmp_path, "r", encoding="utf-8") as f:
            written = f.read()
        if stable_hash(written) != stable_hash(content):
            raise IOError("Hash mismatch after write")

        # Atomic replace — works on all platforms including Windows
        os.replace(tmp_path, path)
        tmp_path = None  # success — no cleanup needed
        return True

    except Exception as e:
        return Error(ErrorCode.FILE_WRITE_FAILED, f"Write failed: {path} → {e}")
    finally:
        # Cleanup: close fd if still open, remove tmp if still exists
        if fd is not None:
            try:
                os.close(fd)
            except OSError:
                pass
        if tmp_path is not None and os.path.exists(tmp_path):
            try:
                os.unlink(tmp_path)
            except OSError:
                pass


def create_snapshot(paths: List[str]) -> Snapshot:
    files: Dict[str, Optional[bytes]] = {}
    for p in paths:
        if os.path.exists(p):
            with open(p, "rb") as f:
                files[p] = f.read()
        else:
            files[p] = None
    return Snapshot(files=files)


def restore_snapshot(snapshot: Snapshot):
    for path, content in snapshot.files.items():
        try:
            if content is None:
                if os.path.exists(path):
                    os.unlink(path)
            else:
                os.makedirs(os.path.dirname(path), exist_ok=True)
                with open(path, "wb") as f:
                    f.write(content)
        except Exception as e:
            LOG.error(f"Rollback failed: {path} → {e}")


def read_clipboard() -> str:
    try:
        if sys.platform == "win32":
            r = subprocess.run(
                ["powershell", "-command", "Get-Clipboard"],
                capture_output=True, text=True, encoding="utf-8", timeout=5,
            )
            return r.stdout.strip() if r.returncode == 0 else ""
        elif sys.platform == "darwin":
            r = subprocess.run(["pbpaste"], capture_output=True, text=True, timeout=5)
            return r.stdout.strip() if r.returncode == 0 else ""
        else:
            r = subprocess.run(
                ["xclip", "-selection", "clipboard", "-o"],
                capture_output=True, text=True, timeout=5,
            )
            return r.stdout.strip() if r.returncode == 0 else ""
    except Exception:
        return ""


def scan_leftover_files(project_dir: str) -> List[str]:
    """Detect .tmp, .bak, orphan files."""
    leftovers = []
    for root, dirs, files in os.walk(project_dir):
        for f in files:
            if f.endswith(".tmp") or f.endswith(".bak"):
                leftovers.append(os.path.join(root, f))
    return leftovers

def execute_transaction(
    ops: List[FileOp],
    project_dir: str,
    dry_run: bool = False,
    diff_mode: bool = False,
) -> TransactionResult:
    tx_id = LOG.new_tx()
    LOG.info(f"Transaction: {len(ops)} operations")

    result = TransactionResult(success=False, tx_id=tx_id, operations=ops)

    if not ops:
        result.success = True
        result.message = "No operations (idempotent)"
        LOG.info(result.message)
        LOG.clear_tx()
        return result

    # DRY RUN
    if dry_run:
        LOG.info("=== DRY RUN ===")
        for i, op in enumerate(ops, 1):
            LOG.info(f"  PLAN [{i}/{len(ops)}]: {op.description}")
        result.success = True
        result.message = "Dry run complete"
        LOG.clear_tx()
        return result

    # DIFF MODE
    if diff_mode:
        LOG.info("=== DIFF MODE ===")
        for op in ops:
            if op.op_type in (FileOpType.WRITE_FILE, FileOpType.UPDATE_INDEX,
                              FileOpType.UPDATE_REGISTRY):
                old = read_file(op.path) or "(not exists)"
                _print_diff(old, op.content, op.path)
        result.success = True
        result.message = "Diff complete"
        LOG.clear_tx()
        return result

    # SNAPSHOT
    affected = [op.path for op in ops if op.op_type != FileOpType.CREATE_DIR]
    snapshot = create_snapshot(affected)

    # EXECUTE
    completed = 0
    try:
        for i, op in enumerate(ops):
            LOG.info(f"  [{i+1}/{len(ops)}] {op.description}")

            if op.op_type == FileOpType.CREATE_DIR:
                os.makedirs(op.path, exist_ok=True)

            elif op.op_type in (FileOpType.WRITE_FILE, FileOpType.UPDATE_INDEX,
                                FileOpType.UPDATE_REGISTRY):
                # Hash check — skip if byte-identical
                existing = read_file(op.path)
                if existing is not None and stable_hash(existing) == stable_hash(op.content):
                    LOG.info("    → No change, skip")
                    result.skipped.append(op.path)
                    continue

                wr = atomic_write(op.path, op.content)
                if isinstance(wr, Error):
                    raise RuntimeError(wr.message)

            completed += 1

    except Exception as e:
        LOG.error(f"FAILED at step {completed+1}: {e}")
        LOG.warning("Rolling back...")
        restore_snapshot(snapshot)
        LOG.warning("Rollback done")
        result.errors.append(Error(ErrorCode.TRANSACTION_FAILED, str(e)))
        result.message = f"Failed & rolled back: {e}"
        LOG.clear_tx()
        return result

    # STATE FINGERPRINT
    fp = compute_project_fingerprint(project_dir)
    result.fingerprint = fp

    result.success = True
    result.message = f"Done: {completed} exec, {len(result.skipped)} skip"
    LOG.info(f"{result.message}")
    LOG.info(f"STATE HASH: {fp}")

    # Leftover check
    leftovers = scan_leftover_files(project_dir)
    if leftovers:
        for lf in leftovers:
            LOG.warning(f"Leftover detected: {lf}")

    LOG.clear_tx()
    return result


def _print_diff(old: str, new: str, path: str):
    print(f"\n{'='*60}\nDIFF: {path}\n{'='*60}")
    if old == new:
        print("  (no changes)")
        return
    ol, nl = old.split("\n"), new.split("\n")
    shown = 0
    for i, (o, n) in enumerate(zip(ol, nl)):
        if o != n and shown < 15:
            print(f"  L{i+1}:\n  --- {o[:100]}\n  +++ {n[:100]}")
            shown += 1
    delta = len(nl) - len(ol)
    if delta > 0:
        print(f"  ... +{delta} lines")
    elif delta < 0:
        print(f"  ... {delta} lines")
    print(f"{'='*60}\n")

def process_input(text: str, project_dir: str) -> TransactionResult:
    """Pipeline: Parse → Validate → Plan → Execute"""
    tx_id = LOG.new_tx()

    # Parse
    parsed = parse_module_block(text)
    if isinstance(parsed, Error):
        LOG.warning(f"Parse rejected: {parsed.message}")
        LOG.clear_tx()
        return TransactionResult(False, tx_id, [], [parsed], message=parsed.message)

    LOG.info(f"Parsed: [{parsed.tag}] group={parsed.group}")

    # Validate
    valid = validate_module(parsed)
    if isinstance(valid, Error):
        LOG.error(f"Validate failed: {valid.message}")
        LOG.clear_tx()
        return TransactionResult(False, tx_id, [], [valid], message=valid.message)

    # Paths
    paths = compute_file_paths(parsed, project_dir)

    # Read state
    reg_json = read_file(paths.registry_file)
    if reg_json:
        try:
            registry = Registry.from_dict(json.loads(reg_json))
        except Exception:
            registry = Registry()
    else:
        registry = Registry()

    index_html = read_file(paths.index_file)
    if not index_html:
        LOG.error("index.html not found")
        LOG.clear_tx()
        return TransactionResult(False, tx_id, [],
                                 [Error(ErrorCode.FILE_MARKER_MISSING, "index.html not found")])

    existing = read_file(paths.module_file)

    # Plan
    ops = generate_file_operations(parsed, paths, registry, index_html, existing)
    if isinstance(ops, Error):
        LOG.error(f"Plan failed: {ops.message}")
        LOG.clear_tx()
        return TransactionResult(False, tx_id, [], [ops], message=ops.message)

    LOG.clear_tx()
    return execute_transaction(ops, project_dir, DRY_RUN, DIFF_MODE)


def initial_build(project_dir: str) -> TransactionResult:
    """Build toàn bộ. Deterministic: cùng code → cùng output byte-level."""
    tx_id = LOG.new_tx()
    LOG.info("=== INITIAL BUILD ===")

    md = os.path.join(project_dir, MODULES_DIR_NAME)
    ad = os.path.join(project_dir, ASSETS_DIR_NAME)

    # Registry — SORTED by tag for determinism
    registry = Registry()
    defaults = [
        ("ABOUT",   "about.js",   gen_about_js),
        ("AST",     "ast.js",     gen_ast_js),
        ("CONTACT", "contact.js", gen_contact_js),
        ("DECOR",   "decor.js",   gen_decor_js),
        ("HOME",    "home.js",    gen_home_js),
        ("TCG",     "tcg.js",     gen_tcg_js),
    ]

    for tag, fname, gen_fn in defaults:
        known = KNOWN_GROUPS[tag]
        content = gen_fn()
        registry.modules[tag] = RegistryEntry(
            tag=tag, group=tag.lower(), filename=fname,
            label=known["label"], icon=known["icon"],
            order=known["order"], content_hash=stable_hash_short(content),
        )

    # Ops
    ops: List[FileOp] = []
    for d in [project_dir, md, ad, os.path.join(ad, "images")]:
        ops.append(FileOp(FileOpType.CREATE_DIR, d, description=f"mkdir {os.path.basename(d)}"))

    core = [
        ("index.html", gen_index_html(registry)),
        ("style.css",  gen_style_css()),
        ("main.js",    gen_main_js()),
    ]
    for name, content in core:
        ops.append(FileOp(
            FileOpType.WRITE_FILE,
            os.path.join(project_dir, name),
            content,
            f"Create {name} ({len(content)}b)",
        ))

    for tag, fname, gen_fn in defaults:
        content = gen_fn()
        ops.append(FileOp(
            FileOpType.WRITE_FILE,
            os.path.join(md, fname),
            content,
            f"Create {fname} ({len(content)}b)",
        ))

    ops.append(FileOp(
        FileOpType.UPDATE_REGISTRY,
        os.path.join(project_dir, CONFIG_FILE_NAME),
        registry.to_json(),
        "Create registry",
    ))

    LOG.clear_tx()
    result = execute_transaction(ops, project_dir, DRY_RUN, DIFF_MODE)

    if result.success and not DRY_RUN and not DIFF_MODE:
        url = f"file:///{os.path.join(project_dir, 'index.html').replace(os.sep, '/')}"
        LOG.info(f"Browser: {url}")
        try:
            webbrowser.open(url)
        except Exception:
            pass

    return result


def verify_consistency(project_dir: str) -> List[Error]:
    errors = []
    rp = os.path.join(project_dir, CONFIG_FILE_NAME)
    rj = read_file(rp)
    if not rj:
        errors.append(Error(ErrorCode.REGISTRY_INCONSISTENT, "Registry missing"))
        return errors
    try:
        reg = Registry.from_dict(json.loads(rj))
    except Exception as e:
        errors.append(Error(ErrorCode.REGISTRY_INCONSISTENT, f"Parse error: {e}"))
        return errors

    md = os.path.join(project_dir, MODULES_DIR_NAME)
    for tag, entry in sorted(reg.modules.items()):
        fp = os.path.join(md, entry.filename)
        if not os.path.exists(fp):
            errors.append(Error(ErrorCode.REGISTRY_INCONSISTENT, f"[{tag}] → missing {entry.filename}"))

    if os.path.exists(md):
        for f in sorted(os.listdir(md)):
            if f.endswith(".js"):
                if not any(e.filename == f for e in reg.modules.values()):
                    errors.append(Error(ErrorCode.REGISTRY_INCONSISTENT, f"Orphan: {f}"))

    leftovers = scan_leftover_files(project_dir)
    for lf in leftovers:
        errors.append(Error(ErrorCode.REGISTRY_INCONSISTENT, f"Leftover: {lf}"))

    return errors


def clipboard_watcher(project_dir: str, interval: float = 0.5):
    last = ""
    LOG.info("=" * 50)
    LOG.info("CLIPBOARD WATCHER ACTIVE")
    LOG.info("  Copy [TAG] + GROUP:xxx + JS body")
    LOG.info("  Ctrl+C to quit")
    LOG.info("=" * 50)

    while True:
        try:
            text = read_clipboard()
            if not text:
                time.sleep(interval)
                continue
            h = stable_hash_short(text)
            if h == last:
                time.sleep(interval)
                continue
            last = h

            if not TAG_PATTERN.match(text.split("\n")[0].strip()):
                continue

            LOG.info(f"Clipboard: {text.split(chr(10))[0].strip()}")
            r = process_input(text, project_dir)
            if r.success:
                LOG.info(f"OK: {r.message}")
                if r.fingerprint:
                    LOG.info(f"STATE HASH: {r.fingerprint}")
            else:
                for e in r.errors:
                    LOG.error(f"  {e.code.name}: {e.message}")

            time.sleep(interval)
        except KeyboardInterrupt:
            LOG.info("Watcher stopped")
            break
        except Exception as e:
            LOG.error(f"Watcher: {e}")
            time.sleep(interval)

class TestStableHash(unittest.TestCase):
    def test_deterministic(self):
        for _ in range(100):
            self.assertEqual(stable_hash("hello"), stable_hash("hello"))

    def test_different(self):
        self.assertNotEqual(stable_hash("a"), stable_hash("b"))

    def test_no_builtin_hash(self):
        """stable_order_key must NOT use builtin hash()"""
        # Run 10 times — builtin hash() changes per process with PYTHONHASHSEED
        results = set()
        for _ in range(10):
            results.add(stable_order_key("NOITHAT_CODIEN"))

class TestInjection(unittest.TestCase):
    def _html(self, scripts=""):
        return f"<html>\n{INDEX_MARKER_START}\n{scripts}\n{INDEX_MARKER_END}\n<script src=\"main.js\"></script>\n</html>"

    def _reg(self, tags):
        d = {}
        for i, t in enumerate(tags):
            kn = KNOWN_GROUPS.get(t, {})
            d[t] = RegistryEntry(t, t.lower(), f"{t.lower()}.js",
                                 kn.get("label", t), kn.get("icon", "x"),
                                 kn.get("order", 100 + i))
        return d

    def test_inject(self):
        r = compute_index_injection(self._html(), self._reg(["HOME", "ABOUT"]))
        self.assertIsInstance(r, str)
        self.assertIn("modules/home.js", r)
        self.assertIn("modules/about.js", r)

    def test_no_duplicate(self):
        r = compute_index_injection(self._html(), self._reg(["HOME"]))
        self.assertEqual(r.count("modules/home.js"), 1)

    def test_missing_marker(self):
        r = compute_index_injection("<html></html>", self._reg(["HOME"]))
        self.assertEqual(r.code, ErrorCode.FILE_MARKER_MISSING)

    def test_dup_marker(self):
        h = f"{INDEX_MARKER_START}\n{INDEX_MARKER_END}\n{INDEX_MARKER_START}\n{INDEX_MARKER_END}"
        self.assertEqual(compute_index_injection(h, self._reg(["HOME"])).code, ErrorCode.FILE_MARKER_DUPLICATE)

    def test_order_independence(self):
        """A,B,C vs C,A,B → same output"""
        h = self._html()
        r1 = compute_index_injection(h, self._reg(["HOME", "ABOUT", "CONTACT"]))
        r2 = compute_index_injection(h, self._reg(["CONTACT", "HOME", "ABOUT"]))
        self.assertEqual(r1, r2)

    def test_idempotent(self):
        h = self._html()
        m = self._reg(["HOME", "ABOUT"])
        r1 = compute_index_injection(h, m)
        r2 = compute_index_injection(r1, m)
        self.assertEqual(r1, r2)
